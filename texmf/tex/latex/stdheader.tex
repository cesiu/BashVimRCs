    \lst@AddToHook{OnNewLine}{
    \let\thelstnumber\relax

}

\newcommand*\ResumeNumber[1]{
    \setcounter{lstnumber}{\numexpr#1-1\relax}
    \lst@AddToHook{OnNewLine}{
        \let\thelstnumber\origthelstnumber
        \refstepcounter{lstnumber}
    }
}
